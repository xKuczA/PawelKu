Opis Projektu

Kalkulator funkcji kwadratowej umożliwia obliczanie miejsc zerowych i wartości funkcji kwadratowej dla podanych współczynników.

Kod Źródłowy

            #include 
            #include 
                
            int main() {
                // Wprowadź współczynniki a, b i c
                double a, b, c;
                std::cout << "Podaj wspolczynniki a, b i c: ";
                std::cin >> a >> b >> c;
                
                // Oblicz deltę
                double delta = b * b - 4 * a * c;
                
               // Sprawdź wartość delty
                if (delta > 0) {
                    // Dwa pierwiastki rzeczywiste
                    double x1 = (-b + std::sqrt(delta)) / (2 * a);
                    double x2 = (-b - std::sqrt(delta)) / (2 * a);
                    std::cout << "Pierwiastki to: " << x1 << " i " << x2 << std::endl;
                } else if (delta == 0) {
                    // Jeden pierwiastek rzeczywisty (podwójny)
                    double x = -b / (2 * a);
                    std::cout << "Pierwiastek to: " << x << std::endl;
                } else {
                    // Brak pierwiastków rzeczywistych
                    std::cout << "Brak pierwiastkow rzeczywistych." << std::endl;
                }
                
                return 0;
            }
